template CheckCommand "my/netapp" {
  import "ipv4-or-ipv6"

  command = [ PluginContribDir + "/check_naf.py", "$netapp_check$" ]

  arguments = {
    "-H" = "$netapp_address$"
    "-C" = {
      value = "$netapp_community$"
      required = true
    }
  }

  vars.netapp_address = "$check_address$"
}

object CheckCommand "my/netapp-status" {
  import "my/netapp"

  vars.netapp_check = "global"
}
